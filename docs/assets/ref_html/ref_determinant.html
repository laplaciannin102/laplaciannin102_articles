$$ \begin{aligned} &i, j \in \{ 1, \cdots , n \}, \quad a_{ij} \in \mathbb{R} \\ \\ &A = \left( \begin{array}{ccc} a_{11} & \cdots & a_{1j} & \cdots & a_{1n} \\ \vdots & & \vdots & & \vdots \\ a_{i1} & \cdots & a_{ij} & \cdots & a_{in} \\ \vdots & & \vdots & & \vdots \\ a_{n1} & \cdots & a_{nj} & \cdots & a_{nn} \end{array} \right) = \left( a_{ij} \right) \\ \\ \\ &\det(A) := \sum_{\sigma \in S_n}{\text{sgn}(\sigma) \prod_{i=1}^{n}{a_{i\sigma(i)}}} \end{aligned} $$
